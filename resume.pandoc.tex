\documentclass[11pt,letterpaper]{moderncv}
% moderncv themes
% style options are 'casual' (default), 'classic', 'banking', 'oldstyle' and 'fancy'
% color options 'black', 'blue' (default), 'burgundy', 'green', 'grey', 'orange', 'purple' and 'red'
\moderncvstyle{banking}
\moderncvcolor{black}

$if(mainfont)$
    \setmainfont[$for(mainfontoptions)$$mainfontoptions$$sep$,$endfor$]{$mainfont$}
$endif$

$if(sansfont)$
     \setsansfont[$for(sansfontoptions)$$sansfontoptions$$sep$,$endfor$]{$sansfont$}
$endif$

$if(monofont)$
     \setmonofont[Mapping=tex-ansi$if(monofontoptions)$,$for(monofontoptions)$$monofontoptions$$sep$ ,$endfor$$endif$]{$monofont$}
$endif$

$if(mathfont)$
     \setmathfont(Digits,Latin,Greek)[$for(mathfontoptions)$$mathfontoptions$$sep$,$endfor$]{$mathfont$}
$endif$

$if(geometry)$
    \usepackage[$for(geometry)$$geometry$$sep$,$endfor$]{geometry}
$endif$

$if(links-as-notes)$
    % Make links footnotes instead of hotlinks:
    \renewcommand{\href}[2]{#2\footnote{\url{#1}}}
$endif$

$for(header-includes)$
$header-includes$
$endfor$

% adjust the page margins
\usepackage[scale=0.75]{geometry}
\usepackage{enumitem}

% adjust itemize
\setlist[itemize]{label={--}, leftmargin=1cm, topsep=.1cm, itemsep=0pt, parsep=0pt}

%% ========================================================================
%% personal data
%% ========================================================================
\name{$firstname$}{$lastname$}

$if(title)$
    \title{$title$}
$endif$

% optional, remove / comment the line if not wanted;
% the "postcode city" and "country" arguments can be omitted or provided empty
$if(address)$
    \address{$if(address.street)$$address.street$$endif$}{$address.city$}{$address.country$}
$endif$

% these are optional and can be commented out
\phone[mobile]{$phone$}
\email{$email$}

$if(homepage)$
    \homepage{$homepage$}
$endif$

$if(linkedin)$
    \social[linkedin]{$linkedin$}
$endif$

$if(twitter)$
    \social[twitter]{$twitter$}
$endif$

$if(github)$
    \social[github]{$github$}
$endif$

$if(summary)$
    \quote{$summary$}
$endif$


%%----------------------------------------------------------------------------------
%% Actual document content
%%----------------------------------------------------------------------------------
\begin{document}
\makecvtitle

$for(include-before)$
$include-before$
$endfor$

\section{Experience}
$for(experience)$
    \cventry{$experience.dates$}{$experience.title$}{$experience.employer$}{$experience.city$}{}{
        $if(experience.description)$
            \begin{itemize}
            $for(experience.description)$
                \item $experience.description$
            $endfor$
            \end{itemize}
        $endif$
    }
$endfor$

\section{Education}
$for(education)$
    \cventry{$education.dates$}{$education.degree$}{$education.institution$}{$education.city$}{}{
        $if(education.description)$
            \begin{itemize}
            $for(education.description)$
                \item $education.description$
            $endfor$
            \end{itemize}
        $endif$
    }
$endfor$

$if(skills)$
\section{Skills}
$for(skills)$
    \cvitem{$skills.name$}{$skills.description$}
$endfor$
$endif$

$if(interests)$
\section{Interests}
$interests$
$endif$

$if(references)$
\section{References}
$references$
$endif$

\end{document}
